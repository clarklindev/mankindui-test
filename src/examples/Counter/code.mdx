```tsx
import { useState } from "react";

import { Counter } from "@mankindui/core";

const CounterExample = () => {
  const [savedData, setSavedData] = useState(0);

  return (
    <Counter
      savedData={savedData}
      onChange={(newValue) => setSavedData(newValue)}
    />
  );
};

export default CounterExample;
```
